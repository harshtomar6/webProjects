  import java.util.Scanner;

  public class student{
    private String usn;
    private String name;
    private String branch;
    private long phone;

    public static  void main(String[] args){
      //Instance of Scanner class
      Scanner input = new Scanner(System.in);

      int n,i;
      
      System.out.println("Enter the value of n");
      n = input.nextInt();

      //Array of objects of type student
      student[] s = new student[n];

      for(i=0;i< n;i++){
        //instance of each object
        s[i] = new student();
        System.out.println("Enter USN of student "+(i+1));
        s[i].usn = input.next();
        System.out.println("Enter name of student "+(i+1));
        s[i].name = input.next();
        System.out.println("Enter branch of student "+(i+1));
        s[i].branch = input.next();
        System.out.println("Enter phone number of student "+(i+1));
        s[i].phone = input.nextLong();
      }

      System.out.println("Details entered by you are:\n");
      for(i=0;i< n;i++){
        System.out.println("Student "+(i+1));
        System.out.println("USN -> "+s[i].usn+"\nName -> "+s[i].name+"\nBranch -> "+s[i].branch+"\nPhone -> "+s[i].phone+"\n");
      }
    }
  }
