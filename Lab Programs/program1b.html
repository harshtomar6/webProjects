  import java.util.Scanner;

  public class Stack{
    //Global variables
    private int stck[] = new int[10];
    private int top = -1;

    public static void main(String[] args){
      //instance of Scanner class
      Scanner input = new Scanner(System.in);
      //instance of Stack class
      Stack s = new Stack();

      int ch,ele;

      while(true){
        System.out.println("==========STACK MENU=========\n");
        System.out.println("1. Push\n2. Pop\n3. Display\n4. Exit\n");
        ch = input.nextInt();

        switch(ch){
        case 1:
          System.out.println("Enter the element");
          ele = input.nextInt();
          s.push(ele);
          System.out.println("Element pushed successfully\n");
          break;
        case 2:
          System.out.println("Element popped is "+s.pop());
          break;
        case 3:
          System.out.println("Stack Contents are:");
          s.display();
          break;
        case 4:
          return;
        }
      }
    }

    public void push(int e){
      stck[++top] = e;
    }

    public int pop(){
      return stck[top--];
    }

    public void display(){
      int i;

      for(i=top;i>=0;i--)
        System.out.println(stck[i]);
    }
}
